mscanB, mscanW, mscanDD:
    input: buffer, bufferSize, result
    output -> result

mprintInt:
    input: buf, bufSize, [<=dd]num
    output: -
    Print dd or lower on the screen.
    Takes num in the 0x16 numeric system

mCLS:
    input: -
    output: -
    Clear screen and set color mode

msetCursor:
    input: [byte]row, [byte]column\
    Set cursor at a position

mprint:
    input: string
    output: -
    Print string on the cursor
